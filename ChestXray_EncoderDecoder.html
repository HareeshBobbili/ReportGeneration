def preProcessImage(img_name):
  img_name = '/content/drive/My Drive/CS2/Data-cs2/NLMCXR_png/'+img_name+'.png'
  #Load image from a file
  image = Image.open(img_name)
  #Convert image into RGB image
  rgb_image = image.convert("RGB")
  #Covert PIL image into numpy array to get the value of each pixel
  pil_array = np.asarray(rgb_image)
  #Centered image to the format the model requires
  pre_process = preprocess_input(pil_array)
  #resizing image to fixed sixe
  re_size = resize(pre_process, (224,224,3))
  #Expands shape of array here it is (1,224,224,3)
  final_image= np.expand_dims(re_size, axis=0)
  return final_image

def covertImageIntovector(dataset):
  features_vector = []
  for frontal , lateral in tqdm(dataset.values,position=0,leave=True):
    #converting images into vector. [tensor shape (1,1024)]
    frontal_vector =  model_features.predict(preProcessImage(frontal))
    lateral_vector =  model_features.predict(preProcessImage(lateral))
    #Concatenate vectors of both images. [tensor shape (1,2048)]
    concat = np.concatenate((frontal_vector,lateral_vector),axis=1)
    #final_conc = np.reshape(concat, concat.shape[1])
    features_vector.append(concat)    
  return features_vector

def createInputOutputPairs(images,report):
  input_pair=[]
  output_pair=[]
  image_pair=[]
  #Converting findings into vector
  report_vector = [token.word_index[word] for word in report.split() if word in token.word_index.keys()]
  for index in range(len(images)):
    for vec in range(1,len(report_vector)):
      #input sequnce
      input = report_vector[:vec]
      #output word for inout sequence
      output = report_vector[vec]
      #post padding on input : size of vector 167
      input_pad = pad_sequences([input],maxlen=max_length,padding='post')[0]
      #converting output word into categorical val : size of vector is vocab size(1545)
      output_cat = tf.keras.utils.to_categorical([output], num_classes=vocab_size)[0]
      input_pair.append(input_pad)
      output_pair.append(output_cat)
      image_pair.append(images[index])

  return np.array(image_pair),np.array(input_pair),np.array(output_pair)

def convertToString(report):
  for index in range(len(report)):
    report[index] = report[index].decode('utf-8')
  return ' '.join(str(word) for word in report)

#Text input layer
input_text = Input(shape=(167,), name='Text')
Embedding_layer = Embedding(input_dim = vocab_size, output_dim = 100, input_length=167, mask_zero=True, trainable=False, 
                weights=[embedding_matrix], name="Embedding_layer")
emb_layer = Embedding_layer(input_text)

#LSTM layer
lstm_layer = LSTM(units=256, activation='tanh', recurrent_activation='sigmoid', use_bias=True,kernel_initializer=tf.keras.initializers.glorot_uniform,recurrent_initializer=tf.keras.initializers.orthogonal,
            bias_initializer=tf.keras.initializers.zeros(), return_sequences=True, name="lstm_layer")(emb_layer)

lstm_layer_1 = LSTM(units=256, activation='tanh', recurrent_activation='sigmoid', use_bias=True,kernel_initializer=tf.keras.initializers.glorot_uniform,recurrent_initializer=tf.keras.initializers.orthogonal,
            bias_initializer=tf.keras.initializers.zeros(), name="lstm_layer_1")
lstm_layer_1_output = lstm_layer_1(lstm_layer)
#Droput 
dropout_layer = Dropout(0.4, name='dropout_layer')(lstm_layer_1_output)

add_layer =  tf.keras.layers.Add()([dense_image, dropout_layer])

#Fully connected layer
fully_connected = Dense(256, activation='relu', kernel_initializer=tf.keras.initializers.he_normal, name='fully_connected')
fc1_output = fully_connected(add_layer)

dropout_layer_1 = Dropout(0.3, name='dropout_layer_1')(fc1_output)
output_layer = Dense(vocab_size, activation='softmax', name='Output_layer')
output = output_layer(dropout_layer_1)

#Final Model
encoder_decoder = Model(inputs = [input_image, input_text], outputs = output)

Loss Function

In [ ]:
crossentropy = tf.keras.losses.CategoricalCrossentropy(from_logits=False, reduction='auto')

def loss_function(y_true, y_pred):
  #Find mask tokens so that we can ignore them while calculating loss
  mask_token = tf.math.logical_not(tf.math.equal(y_true, 0)) 
  #calculating the loss
  loss = crossentropy(y_true, y_pred) 
  #Converting True anf False values into 1's and 0's
  mask_token = tf.cast(mask_token, dtype=loss.dtype)    
  loss = loss*mask_token
  #Mean over all the values
  loss = tf.reduce_mean(loss)
  return loss

Inference Mode

GreedySearch

def greedySearch(image_vec):
  #In the first time step we provide the start token so that the decoder starts generating the next token.
  report = 'start'
  for index in range(max_length):
    input_token = [token.word_index[word] for word in report.split()]
    input_padded = pad_sequences([input_token], maxlen=max_length)
    prediction = encoder_decoder.predict([image_vec,input_padded], verbose=0)
    #greedily select the word with maximum probability
    #selects the most likely word at each step in the output sequence
    prediction = np.argmax(prediction)
    word = token.index_word[prediction]
    report += ' ' + word
    if word == 'end':
      break
  
  finding = report.split()[1:-1]
  return ' '.join(finding)

During inference, the input to the decoder at each time step is the output from the previous time step.

During inference, we generate one word at a time. Thus the Decoder LSTM is called in a loop, every time processing only one time step.

The initial states of the decoder are set to the final states of the encoder.
